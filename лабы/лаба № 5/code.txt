#include <stdio.h>
#include <locale.h>
#include <stdlib.h>



void printSimpleMatrix(int n, float matrix[10][10]) {
    printf("+");
    for (int j = 0; j < n; j++) {
        printf("------");
        if (j < n - 1) printf("+");
    }
    printf("+\n");

    for (int i = 0; i < n; i++) {
        printf("|");
        for (int j = 0; j < n; j++) {
            printf(" %3f  |", matrix[i][j]);
        }
        printf("\n");

        if (i < n - 1) {
            printf("+");
            for (int j = 0; j < n; j++) {
                printf("------");
                if (j < n - 1) printf("+");
            }
            printf("+\n");
        }
    }

    printf("+");
    for (int j = 0; j < n; j++) {
        printf("------");
        if (j < n - 1) printf("+");
    }
    printf("+\n");
}


int main(void) {
    int n;

    printf("Input n: \n");
    scanf_s("%d", &n);

    float arr[10][10];
    double sr_ar = 0;
    int count = 0;

    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            arr[i][j] = rand(); // замени на sacanf_s для нормальной программы
            sr_ar += arr[i][j];
        }
    }
    sr_ar /= n * n;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            if (arr[i][j] > sr_ar) {
                count++;
            }
        }
    }

    printSimpleMatrix(n, arr);
    printf("arithmetic mean - %7.16f\n count of numbers bigger than arithmetic mean - %d", sr_ar, count);
    return 0;
}
