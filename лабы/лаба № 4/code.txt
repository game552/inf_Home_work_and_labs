#include <stdio.h>
#include <stdlib.h>
#include <locale.h>
#include <time.h> // оставь для тестов, а когда показывать будешь убери, а то пиздов дадут

void laba_4() {
    srand((unsigned)time(NULL)); // оставь для тестов, а когда показывать будешь убери, а то пиздов дадут
    double arr[20];
    int min_val = -12;
    int max_val = 13;
    for (int i = 0; i < 20; i++) {
         arr[i] = min_val + rand() % (max_val - min_val + 1);
         }

    for (int i = 0; i < 20; i++) {
        printf("%.2f ", arr[i]);
    }
    printf("\n");

    int check = 1;
    int count = 0;

    for (int k = 0; k < 20; k++) {
        if (arr[k] > 0) {
            count = 0;
            int j1 = k + 1;
            while (j1 < 20 && arr[j1] <= 0) {
                count++;
                j1++;
            }
            if (count > 2) {
                check = 0;
                break;
            }
        }
    }

    if (check) {
        printf("Условие выполнено. Массив без изменений:\n");
        for (int i = 0; i < 20; i++) {
            printf("%.2f ", arr[i]);
        }
        printf("\n");
    }
    else {
        printf("Условие нарушено. Новый массив только из отрицательных элементов:\n");
        double new_arr[20];
        int size = 0;
        for (int i = 0; i < 20; i++) {
            if (arr[i] < 0) {
                new_arr[size++] = arr[i];
            }
        }
        for (int i = 0; i < size; i++) {
            printf("%.2f ", new_arr[i]);
        }
        printf("\n");
    }
}

int main() {
    setlocale(LC_ALL, "Russian");
    laba_4();

    return 0;
}
